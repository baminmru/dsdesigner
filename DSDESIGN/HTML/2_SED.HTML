Запрос:

select K_BSID, CBS, NBS , CBS+' '+ NBS CODE, case WHEN SALDO_TYPE=1 then 'Субконто' else 'Счет' end SALDO_NAME, SALDO_TYPE from K_BS

Запрос строки:

select K_BSID, CBS, NBS, CBS+' '+ NBS CODE, , case WHEN SALDO_TYPE=1 then 'Субконто' else 'Счет' end SALDO_NAME, SALDO_TYPE from K_BS where K_BSID=^K_BSID^

Вставка:

Insert Into K_BS ( CBS, NBS,SALDO_TYPE) Values ( ^CBS^, ^NBS^,^SALDO_TYPE^)

Изменение:

Update K_BS Set CBS=^CBS^, NBS=^NBS^ , SALDO_TYPE=^SALDO_TYPE^ Where K_BSID=^K_BSID^

Удаление:

delete from K_BS where K_BSID=^K_BSID^
